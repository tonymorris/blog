# 404 Error

<div>
<script type="text/javascript">
  var fof = '/404.html';
  if(window.location.pathname != fof) {
    var path = window.location.pathname;
    var p = "/posts";
    var posts = path.substring(0, p.length) == p;
    if(posts)
      window.location = '/404.html';
    else
      window.location = p + path;
  }
</script>
</div>
